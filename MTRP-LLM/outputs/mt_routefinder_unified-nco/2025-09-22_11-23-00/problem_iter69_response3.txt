def heuristics_v2(current_distance_matrix: torch.Tensor, delivery_node_demands: torch.Tensor, current_load: torch.Tensor, delivery_node_demands_open: torch.Tensor, current_load_open: torch.Tensor, time_windows: torch.Tensor, arrival_times: torch.Tensor, pickup_node_demands: torch.Tensor, current_length: torch.Tensor) -> torch.Tensor:

    mutated_distance_scores = -1.3 * (current_distance_matrix ** 1.6) * 0.4 - torch.randn_like(current_distance_matrix) * 0.3

    modified_demand_scores = (current_load.unsqueeze(1) - 1.4 * delivery_node_demands.unsqueeze(0).float()) * 0.4 + \
                             (1.5 / (1 + torch.exp(torch.max(current_load_open) - torch.min(current_load_open))) * 0.7 + torch.randn_like(current_distance_matrix) * 0.4)

    time_score = 0.8 * ((torch.max(arrival_times - time_windows[:, 0].unsqueeze(0), dim=0).values * 0.2) +  
                        (torch.max(time_windows[:, 1].unsqueeze(0) - arrival_times, dim=0).values * 0.8))

    adjusted_pickup_load = current_load.unsqueeze(1) + 1.2 * pickup_node_demands.unsqueeze(0).float()
    pickup_score = (adjusted_pickup_load / (1 + current_distance_matrix) ** 1.4) * 0.15
    
    pickup_score = torch.clamp(pickup_score, min=-float('inf'), max=float('inf'))

    open_delivery_score = (-current_distance_matrix ** 1.6 / (0.7 * delivery_node_demands_open.unsqueeze(0) + 1.4 * torch.mean(current_load_open))).clamp(min=-2.5, max=2.5) * 0.3

    length_score = (current_length.unsqueeze(1) / (2 + current_distance_matrix)).clamp(min=-1, max=1) * 0.25

    overall_scores = mutated_distance_scores + modified_demand_scores - time_score + pickup_score + open_delivery_score + length_score

    return overall_scores
