import torch
def heuristics_v2(current_distance_matrix: torch.Tensor, delivery_node_demands: torch.Tensor, current_load: torch.Tensor, delivery_node_demands_open: torch.Tensor, current_load_open: torch.Tensor, time_windows: torch.Tensor, arrival_times: torch.Tensor, pickup_node_demands: torch.Tensor, current_length: torch.Tensor) -> torch.Tensor:

    mutated_distance_scores = -0.9 * (current_distance_matrix ** 1.4) * 0.4 - torch.randn_like(current_distance_matrix) * 0.6  # Changing the exponent and weights in mutated distance scores

    modified_demand_scores = ((current_load.unsqueeze(1) - delivery_node_demands.unsqueeze(0).float()) ** 0.8) * 0.3 - torch.mean(current_load)  # Modified demand score calculation

    time_score = 1.2 * ((torch.max(arrival_times - time_windows[:, 0].unsqueeze(0), dim=0).values * 0.2) +
                        (torch.max(time_windows[:, 1].unsqueeze(0) - arrival_times, dim=0).values * 0.8))

    adjusted_pickup_load = current_load.unsqueeze(1) + pickup_node_demands.unsqueeze(0).float()
    pickup_score = ((adjusted_pickup_load / (1 + torch.sqrt(current_distance_matrix))) ** 2) * 0.25  # Modified pickup score calculation

    pickup_score = torch.clamp(pickup_score, min=-float('inf'), max=float('inf'))

    open_delivery_score = (-current_distance_matrix ** 1.1 / (2 + delivery_node_demands_open.unsqueeze(0) + torch.mean(current_load_open)*2)).clamp(min=-1, max=1) * 0.2  # Adjusted open delivery score calculation

    length_score = (current_length.unsqueeze(1) / (1 + current_distance_matrix) ** 0.7).clamp(min=-1, max=1) * 0.35  # Changed exponential power and weights in length score

    overall_scores = mutated_distance_scores + modified_demand_scores - time_score + pickup_score + open_delivery_score + length_score

    return overall_scores
