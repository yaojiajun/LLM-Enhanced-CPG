    time_score = 0.7 * ((torch.max(arrival_times - time_windows[:, 0].unsqueeze(0), dim=0).values * 0.3) +

    adjusted_pickup_load = current_load.unsqueeze(1) + pickup_node_demands.unsqueeze(0).float()
    pickup_score = (adjusted_pickup_load / (1 + current_distance_matrix) ** 2) * 0.2 * (1 - (delivery_node_demands.unsqueeze(0) / (current_load.unsqueeze(1) + 1)))  # Modified logic for pickup score

    pickup_score = torch.clamp(pickup_score, min=-float('inf'), max=float('inf'))

    open_delivery_score = (-current_distance_matrix ** 1.2 / (delivery_node_demands_open.unsqueeze(0) + torch.mean(current_load_open)*1.2)).clamp(min=-2, max=2) * 0.2  # Adjusted load balancing factor in open_delivery_score

    length_score = (current_length.unsqueeze(1) / (1 + current_distance_matrix)).clamp(min=-1, max=1) * 0.1

    overall_scores = mutated_distance_scores + modified_delivery_scores - time_score + delivery_node_demands_unload_score + open_node_unload_score + length_score

    return overall_scores
