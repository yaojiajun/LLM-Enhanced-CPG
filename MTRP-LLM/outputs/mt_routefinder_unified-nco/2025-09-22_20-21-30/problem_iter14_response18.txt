import torch

def heuristics_v2(current_distance_matrix: torch.Tensor, delivery_node_demands: torch.Tensor, current_load: torch.Tensor, delivery_node_demands_open: torch.Tensor, current_load_open: torch.Tensor, time_windows: torch.Tensor, arrival_times: torch.Tensor, pickup_node_demands: torch.Tensor, current_length: torch.Tensor) -> torch.Tensor:

    mutated_distance_scores = -1 * (current_distance_matrix ** 1.2) * 0.3 - torch.randn_like(current_distance_matrix) * 0.2

    modified_demand_scores = (current_load.unsqueeze(1) - delivery_node_demands.unsqueeze(0).float()) * 0.2 + \
                             (1 / (1 + torch.exp(torch.mean(current_load_open) - torch.min(current_load_open))) * 0.4 + torch.randn_like(current_distance_matrix) * 0.2)

    time_score = 0.7 * ((torch.max(arrival_times - time_windows[:, 0].unsqueeze(0), dim=0).values * 0.3) +
                        (torch.max(time_windows[:, 1].unsqueeze(0) - arrival_times, dim=0).values * 0.7))

    adjusted_pickup_load = current_load.unsqueeze(1) + pickup_node_demands.unsqueeze(0).float()
    pickup_score = (adjusted_pickup_load / (1 + current_distance_matrix) ** 1.6) * 0.17 * (1 - (delivery_node_demands.unsqueeze(0) / (current_load.unsqueeze(1) + 1)) ** 1.2)

    open_delivery_score = torch.abs(current_load_open.unsqueeze(1) - delivery_node_demands_open) * 0.15  # Modified weight for open delivery score (turned into an absolute value)
    delivery_score_mod = (-current_distance_matrix ** 1.2 / (delivery_node_demands_open.unsqueeze(0) * 1.5 + current_load_open.unsqueeze(1) * 0.8)).clamp(min=-2, max=2) * 0.25
    delivery_score_mod += open_delivery_score  # Include the modified open delivery score in the total delivery score calculation

    length_score = (current_length.unsqueeze(1) - current_distance_matrix) * 0.22 / (1 + current_distance_matrix)
    length_score = length_score.clamp(min=-0.5, max=0.5) * 0.3

    overall_scores = mutated_distance_scores + modified_demand_scores - time_score + pickup_score + delivery_score_mod + length_score

    return overall_scores
