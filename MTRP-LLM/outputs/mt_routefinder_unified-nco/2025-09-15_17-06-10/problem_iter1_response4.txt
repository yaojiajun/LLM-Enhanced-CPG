```python
import torch
import torch.nn.functional as F

def heuristics_v2(current_distance_matrix, delivery_node_demands, current_load, delivery_node_demands_open, current_load_open, time_windows, arrival_times, pickup_node_demands, current_length):
    # Calculate scores based on various criteria
    load_scores = current_distance_matrix * (current_load - delivery_node_demands)
    open_load_scores = current_distance_matrix * (current_load_open - delivery_node_demands_open)
    time_scores = F.relu(arrival_times - time_windows[:, 1].unsqueeze(0)) / arrival_times
    pickup_scores = current_distance_matrix * pickup_node_demands
    length_scores = torch.abs(current_distance_matrix - current_length)

    # Combine scores
    total_scores = load_scores + open_load_scores + time_scores + pickup_scores + length_scores

    return total_scores
```
