def heuristics_v2(current_distance_matrix: torch.Tensor, delivery_node_demands: torch.Tensor, current_load: torch.Tensor, delivery_node_demands_open: torch.Tensor, current_load_open: torch.Tensor, time_windows: torch.Tensor, arrival_times: torch.Tensor, pickup_node_demands: torch.Tensor, current_length: torch.Tensor) -> torch.Tensor:

    # Generate a mutation in the distance-based heuristic score matrix with different penalty weights and randomness
    mutated_distance_scores = -1 * (current_distance_matrix ** 1.2) * 0.15 + torch.randn_like(current_distance_matrix) * 0.25

    # Generate the original demand-based heuristic score matrix with the same calculations as in heuristics_v1
    original_demand_scores = (current_load.unsqueeze(1) - delivery_node_demands.unsqueeze(0).float()) * 0.3 + (1 / (1 + torch.exp(torch.mean(current_load) - torch.min(current_load)))) + torch.randn_like(current_distance_matrix) * 0.3

    # Calculate the overall score matrix considering the mutation in distance scores and original demand scores
    overall_scores = mutated_distance_scores + original_demand_scores

    return overall_scores
