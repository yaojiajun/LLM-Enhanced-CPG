import torch

def heuristics_v2(current_distance_matrix: torch.Tensor, delivery_node_demands: torch.Tensor, current_load: torch.Tensor, delivery_node_demands_open: torch.Tensor, current_load_open: torch.Tensor, time_windows: torch.Tensor, arrival_times: torch.Tensor, pickup_node_demands: torch.Tensor, current_length: torch.Tensor) -> torch.Tensor:
    
    # Modify the calculation for the heuristic scores based on 'current_distance_matrix', 'delivery_node_demands', 'current_load' inputs
    distance_heuristic = 1.0 / (current_distance_matrix + 1e-8)  # Adjusted distance heuristic
    delivery_score = delivery_node_demands / (current_load + 1e-8)  # Adjusted delivery score
    pickup_score = pickup_node_demands / (current_load + 1e-8)  # Adjusted pickup score
    
    total_score = distance_heuristic + delivery_score - pickup_score
    
    return total_score
