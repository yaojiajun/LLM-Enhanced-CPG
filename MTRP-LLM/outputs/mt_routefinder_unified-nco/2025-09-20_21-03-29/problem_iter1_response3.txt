    distance_heuristic = torch.exp(-current_distance_matrix)  # Example modified distance heuristic calculation
    delivery_score = delivery_node_demands / (current_load + 1e-8) + torch.rand(delivery_node_demands.shape)  # Example modified delivery score calculation
    pickup_score = pickup_node_demands / (current_load + 1e-8) + torch.rand(pickup_node_demands.shape)  # Example modified pickup score calculation
