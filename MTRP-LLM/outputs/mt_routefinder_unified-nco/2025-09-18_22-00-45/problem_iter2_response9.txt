def heuristics_v2(current_distance_matrix, delivery_node_demands, current_load, delivery_node_demands_open, current_load_open, time_windows, arrival_times, pickup_node_demands, current_length):
    # Compute heuristic scores based on load constraints, time windows, pickup demands, and route length constraints
    load_diff = delivery_node_demands - current_load
    demand_diff = pickup_node_demands - current_load
    time_diff = arrival_times[:, 1] - arrival_times[:, 0]
